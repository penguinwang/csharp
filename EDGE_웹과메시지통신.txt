namespace PenguinApp
{
    public partial class Form1 : Form
    {
        public Form1()
        {
            InitializeComponent();
        }

        private async void Form1_Load(object sender, EventArgs e)
        {
            await this.webView21.EnsureCoreWebView2Async(null);
            this.webView21.Source = new System.Uri("http://wizupdate.com/csharp.html", System.UriKind.Absolute);
        }

        private void button1_Click(object sender, EventArgs e)
        {
            if (webView21 != null && webView21.CoreWebView2 != null)
            {
                webView21.CoreWebView2.ExecuteScriptAsync("CallScript(\"아름다운\", \"KOREA\");");  // 함수호출
            }
        }

        private void webView21_WebMessageReceived(object sender, Microsoft.Web.WebView2.Core.CoreWebView2WebMessageReceivedEventArgs e)
        {
            string s = e.TryGetWebMessageAsString();
            // MessageBox.Show(s);
            string[] ss = s.Split(new char[] { '&' });
            if (ss[0] == "user_def")
            {
                user_def(s);
            }
        }
        private void user_def(string parm)
        {
            string[] ss = parm.Split(new char[] { '&' });
            MessageBox.Show("sended from Script : " + ss[1]);
        }

        private async void button2_Click(object sender, EventArgs e)
        {
            string result = await webView21.ExecuteScriptAsync("document.getElementById('text1').value") ;    // reading
            MessageBox.Show(result.Replace("\"",""));  // 값을 읽었을 때 따옴표가 붙어서 제거
            await webView21.ExecuteScriptAsync("document.getElementById('text2').value=\"아름다운 KOREA\";"); // writeing
        }
    }
}
